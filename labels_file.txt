    def pad(self, batch: Batch):
        """Pad a given batch by filling a given pad value."""
        max_length = max(len(example) for example in batch)
        padded = []
        for example in batch:
            example += ([-100] * (max_length - len(example)))
            padded.append(example)
        return padded
